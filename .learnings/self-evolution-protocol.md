```python
class AccuracyTracker:
    """准确率追踪器"""
    
    def __init__(self):
        self.predictions = []
        self.results = []
    
    def record_prediction(self, prediction):
        """记录预测"""
        self.predictions.append({
            'time': now(),
            'content': prediction,
            'verified': False
        })
    
    def verify_prediction(self, prediction_id, actual_result):
        """验证预测"""
        prediction = self.predictions[prediction_id]
        prediction['verified'] = True
        prediction['actual'] = actual_result
        prediction['correct'] = (prediction['content'] == actual_result)
        
        # 计算准确率
        return self.calculate_accuracy()
    
    def calculate_accuracy(self):
        """计算准确率"""
        verified = [p for p in self.predictions if p['verified']]
        if not verified:
            return None
        correct = [p for p in verified if p['correct']]
        return len(correct) / len(verified)
```

|------|--------|--------|----------|
| S级 | 官方数据、交易所数据 | 99%+ | 核心决策 |
| A级 | 主流财经媒体、权威机构 | 95%+ | 辅助决策 |
